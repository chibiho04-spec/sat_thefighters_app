# 受注番号の後変更（リネーム）機能 実装計画

> **For agentic workers:** この計画は単一HTMLファイル `案件管理アプリ/index.html` を編集する。自動テスト基盤は無いため、各タスクの検証は (a) `node --check` 相当のJS構文チェック、(b) Chrome実機E2E（同期は `window._gasUrlOne=()=>''` で無効化し本番シート保護）で行う。ステップはチェックボックスで追跡。

**Goal:** 登録後ロックされている受注番号を、データ管理画面（案件＋単独請求書一覧）と請求書編集画面から後で変更でき、紐づく請求書も追従し、重複時は3択警告で解決する。

**Architecture:** 中核は純粋関数 `renameOrderNum(oldNum,newNum)`（全キー・参照・同期・画面を整合更新）。その上に衝突/請求済みを捌くオーケストレータ `_applyOrderRename()`、入力UI（データ管理セクション＋請求書編集の✏️）を載せる。衝突解決は3ボタンのカスタムモーダル。

**Tech Stack:** 素のHTML/CSS/JS（インライン）、localStorage、Googleスプレッドシート同期(pushKind)。

参照仕様：`docs/superpowers/specs/2026-06-17-受注番号-後変更-design.md`

## 既存コードの要点（実装時に該当行を再確認）

- `ws_${num}` / `inv_${num}` が主キー。`saveInvStatus(num,data)`(≈8910) は `inv_${num}` 書込＋`pushKind('invoice',[Object.assign({'受注No':num},data)])`。`loadInvStatus(num)`(≈8907)。
- `getSavedProjects()`(≈14657) / `projects_saved`。各要素に `.num`。
- `_orderNumExists(num)`(≈8731)：ws_/inv_/wsData/projects を走査して重複判定（流用する）。
- 同期 pushKind パターン：
  - order 作成 `pushKind('order',[{'受注No':num,_project:wsData[num]||{},_updatedAt:iso}])`
  - order 削除 `pushKind('order',[{'受注No':num,_project:{},_updatedAt:iso,_deleted:true}])`
  - child 作成 `pushKind('child',[Object.assign({},data,{'受注No':num})])`
  - child 削除 `pushKind('child',[{'受注No':num,_updatedAt:iso,_deleted:true}])`
  - invoice 作成 `pushKind('invoice',[Object.assign({'受注No':num},data)])`（saveInvStatus が実施）
  - invoice 削除 `pushKind('invoice',[{'受注No':num,_deleted:true,_updatedAt:iso}])`
- 請求書編集の番号欄：`inv-order-edit`(編集) / `inv-order-readonly`(表示)、`_setInvOrderField(num,editable)`(≈8748)。
- ワークシート表示番号：`ws-detail-num`(≈6258)。請求書表示番号：`inv-detail-num`。
- データ管理画面：バックアップ/復元/リセットのある画面（実装時に id を grep して挿入位置特定）。

```js
  // 0000 から空いている退避用番号を返す（0000,0001,...）。衝突相手を逃がす先。
  function _freeZeroNum() {
    for (let i = 0; i < 10000; i++) {
      const cand = String(i).padStart(4, '0');
      if (!(typeof _orderNumExists === 'function' && _orderNumExists(cand))) return cand;
    }
    return '0000';
  }

  // 受注番号(主キー)を oldNum→newNum にリネーム。newNum は呼び出し前に「空き」である前提。
  // ws_/inv_キー・projects_saved・wsData/invData・削除リスト・同期・画面表示を整合更新する。
  function renameOrderNum(oldNum, newNum) {
    oldNum = String(oldNum); newNum = String(newNum);
    if (!newNum || oldNum === newNum) return false;
    const nowIso = new Date().toISOString();
    const syncOn = (typeof _gasUrlOne === 'function' && _gasUrlOne() && typeof pushKind === 'function');

    // 1. ws_ キー移行
    let wsContent = null;
    const wsRaw = localStorage.getItem('ws_' + oldNum);
    if (wsRaw !== null) {
      localStorage.setItem('ws_' + newNum, wsRaw);
      localStorage.removeItem('ws_' + oldNum);
      try { wsContent = JSON.parse(wsRaw); } catch (e) { wsContent = {}; }
    }

    // 2. inv_ キー移行（saveInvStatus が inv_newNum 書込＋invoice同期(new)を実施）
    let invContent = null;
    const invRaw = localStorage.getItem('inv_' + oldNum);
    if (invRaw !== null) {
      try { invContent = JSON.parse(invRaw) || {}; } catch (e) { invContent = {}; }
      localStorage.removeItem('inv_' + oldNum);
      if (typeof saveInvStatus === 'function') saveInvStatus(newNum, invContent);
      else localStorage.setItem('inv_' + newNum, invRaw);
    }

    // 3. projects_saved
    try {
      const list = JSON.parse(localStorage.getItem('projects_saved') || '[]');
      const i = list.findIndex(p => String(p.num) === oldNum);
      if (i >= 0) { list[i].num = newNum; list[i]._updatedAt = nowIso; localStorage.setItem('projects_saved', JSON.stringify(list)); }
    } catch (e) { console.warn('projects rename', e); }

    // 4. メモリ wsData / invData
    try { if (typeof wsData === 'object' && wsData[oldNum]) { wsData[newNum] = wsData[oldNum]; if (wsData[newNum] && typeof wsData[newNum] === 'object') wsData[newNum].num = newNum; delete wsData[oldNum]; } } catch (e) {}
    try { if (typeof invData === 'object' && invData && invData[oldNum]) { invData[newNum] = invData[oldNum]; delete invData[oldNum]; } } catch (e) {}

    // 5. 削除リスト（通常は無いが防御）
    ['deleted_ws', 'deleted_inv'].forEach(k => {
      try { let a = JSON.parse(localStorage.getItem(k) || '[]'); if (a.indexOf(oldNum) >= 0) { a = a.map(x => x === oldNum ? newNum : x); localStorage.setItem(k, JSON.stringify(a)); } } catch (e) {}
    });

    // 6. 同期：旧番号を削除フラグ、新番号を作成（二重行防止）
    if (syncOn) {
      try {
        pushKind('order', [{ '受注No': oldNum, _project: {}, _updatedAt: nowIso, _deleted: true }]).catch(() => {});
        pushKind('child', [{ '受注No': oldNum, _updatedAt: nowIso, _deleted: true }]).catch(() => {});
        if (invContent) pushKind('invoice', [{ '受注No': oldNum, _deleted: true, _updatedAt: nowIso }]).catch(() => {});
        if (wsContent !== null) {
          const wsNew = (typeof wsData === 'object' && wsData[newNum]) ? wsData[newNum] : (wsContent || {});
          pushKind('order', [{ '受注No': newNum, _project: wsNew, _updatedAt: nowIso }]).catch(() => { try { pendingAdd('order', newNum); } catch (_) {} });
          pushKind('child', [Object.assign({}, wsContent, { '受注No': newNum, _updatedAt: nowIso })]).catch(() => { try { pendingAdd('child', newNum); } catch (_) {} });
        }
      } catch (e) { console.warn('rename sync', e); }
    }

    // 7. 画面反映
    try { const wd = document.getElementById('ws-detail-num'); if (wd && wd.textContent === oldNum) wd.textContent = newNum; } catch (e) {}
    try { const idn = document.getElementById('inv-detail-num'); if (idn && idn.textContent === oldNum) idn.textContent = newNum; } catch (e) {}
    try { if (typeof renderHomeCal === 'function') renderHomeCal(); } catch (e) {}
    try { if (typeof renderSalesScreen === 'function') renderSalesScreen(); } catch (e) {}
    return true;
  }
```

- [ ] **Step 2: オーケストレータ `_applyOrderRename` を追加**（形式チェック→請求済み確認→衝突3択→renameOrderNum）

```js
  // 受注番号変更の入口。oldNum を newNumRaw に変更する。成功で確定後の番号、未実行は null を返す。
  async function _applyOrderRename(oldNum, newNumRaw) {
    oldNum = String(oldNum);
    // 形式：英数字とハイフンのみ（既存ルール）。大文字化。
    let newNum = String(newNumRaw || '').toUpperCase().replace(/[^0-9A-Z-]/g, '');
    if (!newNum) { alert('番号を入力してください。'); return null; }
    if (newNum === oldNum) return null; // 変更なし

    // 請求済み/入金済み確認
    try {
      const st = (typeof loadInvStatus === 'function') ? loadInvStatus(oldNum) : {};
      if (st && (st.billed || st.paid)) {
        if (!confirm('この案件は請求済み／入金済みです。\n客に出した受注番号が変わります。本当に変更しますか？')) return null;
      }
    } catch (e) {}

    // 衝突チェック
    if (typeof _orderNumExists === 'function' && _orderNumExists(newNum)) {
      // 相手の件名を取得
      let bTitle = '';
      try {
        const p = JSON.parse(localStorage.getItem('projects_saved') || '[]').find(x => String(x.num) === newNum);
        if (p) bTitle = p.title || '';
        if (!bTitle) { const inv = (typeof loadInvStatus === 'function') ? loadInvStatus(newNum) : {}; bTitle = (inv && inv.title) || ''; }
      } catch (e) {}
      const choice = await _orderConflictDialog(newNum, newNum, bTitle);
      if (choice === 'cancel') return null;
      if (choice === 'a') {            // こちら優先：相手を0000へ逃がしてから自分を newNum に
        renameOrderNum(newNum, _freeZeroNum());
        renameOrderNum(oldNum, newNum);
        return newNum;
      }
      if (choice === 'b') {            // 相手優先：自分を0000へ
        const z = _freeZeroNum();
        renameOrderNum(oldNum, z);
        return z;
      }
      return null;
    }

    // 衝突なし
    renameOrderNum(oldNum, newNum);
    return newNum;
  }
```

  // データ管理：受注番号変更の一覧（案件＋単独請求書）を描画
  function renderOrderNumEditList() {
    const box = document.getElementById('ordernum-edit-list');
    if (!box) return;
    const delWs = (typeof getDeletedWS === 'function') ? getDeletedWS() : [];
    let delInv = []; try { delInv = JSON.parse(localStorage.getItem('deleted_inv') || '[]'); } catch (e) {}
    const rows = []; const seen = new Set();
    // 案件
    try {
      JSON.parse(localStorage.getItem('projects_saved') || '[]').forEach(p => {
        const num = String(p.num || ''); if (!num || delWs.indexOf(num) >= 0 || seen.has(num)) return; seen.add(num);
        rows.push({ num, title: p.title || '', sub: p.company || '' });
      });
    } catch (e) {}
    // 単独請求書（projects に無い inv_）
    Object.keys(localStorage).forEach(k => {
      if (k.indexOf('inv_') !== 0) return; const num = k.slice(4);
      if (seen.has(num) || delInv.indexOf(num) >= 0) return; seen.add(num);
      let t = ''; try { const s = JSON.parse(localStorage.getItem(k) || '{}'); t = s.title || s.company || ''; } catch (e) {}
      rows.push({ num, title: t, sub: '（単独請求書）' });
    });
    const esc = (typeof escapeHtml === 'function') ? escapeHtml : (s => String(s));
    box.innerHTML = rows.map(r => {
      const zero = /^0\d{3}$/.test(r.num);
      return `<div style="display:flex;align-items:center;gap:8px;padding:7px 4px;border-bottom:1px dashed #ddd;">
        <span style="font-family:'DotGothic16',monospace;font-weight:bold;font-size:14px;color:${zero ? '#c0392b' : '#2c3e50'};min-width:84px;">${esc(r.num)}${zero ? ' ⚠' : ''}</span>
        <span style="flex:1;min-width:0;font-size:12px;color:#555;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${esc(r.title)} <span style="color:#aaa;">${esc(r.sub)}</span></span>
        <button onclick="startOrderNumEdit('${esc(r.num)}')" style="font-family:'DotGothic16',monospace;font-size:12px;background:#f1c40f;color:#5d4037;border:2px solid #b7950b;border-radius:4px;padding:5px 10px;cursor:pointer;white-space:nowrap;">✏️変更</button>
      </div>`;
    }).join('') || '<div style="color:#999;font-size:12px;padding:8px;">対象がありません</div>';
  }

  // ✏️変更：新番号を尋ねて _applyOrderRename を呼ぶ
  async function startOrderNumEdit(oldNum) {
    const cur = String(oldNum);
    const input = prompt('新しい受注番号を入力（接頭辞 T/E/P ＋ 数字。例 T26050）\n現在：' + cur, cur);
    if (input == null) return;
    const result = await _applyOrderRename(cur, input);
    if (result) { renderOrderNumEditList(); alert('受注番号を「' + result + '」に変更しました。'); }
  }
```

  // 請求書編集画面：単独請求書(ws_が無い)のときだけ「✏️変更」を出す
  function _maybeShowInvOrderEditBtn(num) {
    const btn = document.getElementById('inv-order-edit-btn');
    if (!btn) return;
    const hasWs = !!localStorage.getItem('ws_' + num);
    btn.style.display = (num && !hasWs) ? 'inline-block' : 'none';
    btn.onclick = async () => {
      const result = await _applyOrderRename(num, prompt('新しい受注番号（例 T26050）\n現在：' + num, num) || '');
      if (result) {
        const idn = document.getElementById('inv-detail-num'); if (idn) idn.textContent = result;
        if (typeof _setInvOrderField === 'function') _setInvOrderField(result, false);
        // 編集画面が result を指すよう内部状態も更新（_invCurrentNum 等があれば合わせる）
        alert('受注番号を「' + result + '」に変更しました。');
      }
    };
  }
```
