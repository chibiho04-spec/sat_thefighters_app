# 請求書テンプレート機能 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 定期案件の請求書を「ひな形（テンプレート）」として保存し、そこから受注番号自動・日付/数量/金額空の新規請求書をワンタッチで起こせるようにする。

**Architecture:** 単一HTMLファイル `index.html` にインラインで追加。テンプレートは localStorage `inv_templates`（配列）にこの端末のみ保存（同期・売上に非関与）。作成フローは既存 `_openBlankInvScreen`（新規番号採番＋重複チェック）の初期化を流用し、その上にテンプレ内容を流し込む。明細収集は既存 `saveInvDoc`/`createDeliveryFromCurrent` と共通化（DRY）。

**Tech Stack:** プレーン HTML/CSS/JS（フレームワークなし）。自動テストランナーは無いため、検証は (1) `node` による `<script>` 構文チェック、(2) ローカルサーバ `python3 -m http.server` + ブラウザ console での挙動確認（`window.__x` に結果を入れてポーリング）で行う。

  function loadInvTemplates() {
    try { return JSON.parse(localStorage.getItem('inv_templates')) || []; } catch (e) { return []; }

  function saveInvTemplates(list) {
    localStorage.setItem('inv_templates', JSON.stringify(Array.isArray(list) ? list : []));

  // 編集画面の明細行を収集（saveInvDoc / createDeliveryFromCurrent / テンプレ保存で共通利用）
  function _collectInvLines() {
    const tbody = document.getElementById('inv-lines-body');
    const lines = [];
    if (!tbody) return lines;
    tbody.querySelectorAll('.inv-line-row').forEach(tr => {
      const nameEl  = tr.querySelector('.inv-line-name') || tr.querySelector('.inv-td-name span');
      const unitEl  = tr.querySelector('.inv-line-unit') || tr.querySelector('.inv-td-unit span');
      const qtyEl   = tr.querySelector('.inv-line-qty')  || tr.querySelector('.inv-td-qty span');
      const priceEl = tr.querySelector('.inv-line-price');
      const priceTextEl = tr.querySelector('.inv-td-price span');
      const codeEl  = tr.querySelector('.inv-td-code');
      const taxEl   = tr.querySelector('.inv-line-tax');
      const outsourceEl = tr.querySelector('.inv-line-outsource');
      const name  = (nameEl?.value  || nameEl?.textContent  || '').trim();
      const unit  = (unitEl?.value  || unitEl?.textContent  || '').trim();
      const qty   = Number((qtyEl?.value || qtyEl?.textContent || '').replace(/[^\d.]/g, '')) || 0;
      const rawPrice = (priceEl?.value || (priceTextEl?.textContent || '').replace(/[¥,]/g, ''));
      const price = Number(rawPrice) || 0;
      const code  = (codeEl?.textContent || '').trim();
      const taxable = taxEl ? (taxEl.value === '1') : true;
      const outsource = outsourceEl ? !!outsourceEl.checked : false;
      if (!name && !price) return;
      lines.push({ code, name, qty, unit, price, taxable, outsource });
    });
    return lines;
  }
```

  // 編集画面のヘッダー＋明細＋外注費＋値引きをまとめて収集（テンプレ保存／納品書化で共通利用）
  function _collectInvEditorContent() {
    return {
      company:  (document.getElementById('inv-screen-company')?.value || '').trim(),
      contact:  (document.getElementById('inv-screen-contact')?.value || '').trim(),
      honorific: document.getElementById('inv-screen-honorific')?.value || '御中',
      zip:      (document.getElementById('inv-screen-zip')?.textContent  || '').trim(),
      addr:     (document.getElementById('inv-screen-addr')?.textContent || '').trim(),
      title:    document.getElementById('inv-screen-title')?.value || '',
      staff:    (document.getElementById('inv-screen-staff')?.value || '').trim(),
      workDate: document.getElementById('inv-screen-workdate')?.value || '',
      memo:     (document.getElementById('inv-screen-memo')?.value || '').trim(),
      docMode:  _docMode || 'invoice',
      lines:    _collectInvLines(),
      outsourceList: (typeof getInvOutsourceList === 'function') ? getInvOutsourceList() : [],
      discount: parseFloat(document.getElementById('inv-discount')?.value || 0) || 0
    };
  }

  // 現在の請求書編集内容を「ひな形」として保存する。数量・単価・外注金額・値引きは空にする。
  function saveCurrentAsTemplate() {
    const c = _collectInvEditorContent();
    if (!c.lines.length && !c.company && !c.title) {
      alert('保存できる内容がありません。先に請求書の内容を入力するか、既存の請求書を開いてください。');
      return;
    }
    const defname = c.title ? (c.title + (c.company ? '（' + c.company + '）' : '')) : (c.company || '無題のテンプレート');
    const name = (prompt('テンプレート名を入力してください', defname) || '').trim();
    if (!name) return; // キャンセル
    const tpl = {
      id: 'tpl_' + Date.now(),
      name: name,
      company: c.company, contact: c.contact, honorific: c.honorific,
      zip: c.zip, addr: c.addr, title: c.title, staff: c.staff, memo: c.memo,
      docMode: c.docMode,
      // 品目名・単位・区分は残し、数量・単価は空にする
      lines: c.lines.map(l => ({ code: l.code || '', name: l.name || '', unit: l.unit || '',
                                 qty: '', price: '', taxable: l.taxable !== false, outsource: !!l.outsource })),
      // 外注先名は残し、金額は空に
      outsourceList: (Array.isArray(c.outsourceList) ? c.outsourceList : []).map(o => ({
                                 name: o.name || '', cost: '', taxable: !!o.taxable })),
      _updatedAt: new Date().toISOString()
    };
    const list = loadInvTemplates();
    list.push(tpl);
    saveInvTemplates(list);
    const bubble = document.querySelector('.speech-bubble');
    if (bubble) bubble.innerHTML = `📑 テンプレート「${escapeHtml(name)}」を保存しました<span class="blink">█</span>`;
    alert('📑 テンプレート「' + name + '」を保存しました。\n会計室の「請求書テンプレート」から呼び出せます。');
  }
```
