function applyCompanyConfig() {
  const C = COMPANY_CONFIG;
  document.title = C.appTitle;
  // ログインロゴ（疑似要素のcontentを上書き）
  const st = document.createElement('style');
  st.textContent = `.login-logo::before{content:'${C.appTitleLogo[0]}\\A${C.appTitleLogo[1]}' !important;}`;
  // ※実際のセレクタは元CSS（行2029付近）のセレクタ名に合わせること
  document.head.appendChild(st);
  // 発行者情報
  const set = (id, v) => { const el = document.getElementById(id); if (el) el.textContent = v; };
  set('issuer-name', C.company.name);
  set('issuer-address', C.company.address);
  set('issuer-tel', C.company.tel);
  const reg = document.getElementById('issuer-reg');
  if (reg) { if (C.company.invoiceReg) reg.textContent = '登録番号 ' + C.company.invoiceReg; else reg.style.display = 'none'; }
  const banks = document.getElementById('issuer-banks');
  if (banks) banks.innerHTML = C.company.banks.map(b => `<div>${escapeHtml(b)}</div>`).join('');
}
```

    for (let i = 0; i < localStorage.length; i++) {
      const k = localStorage.key(i);
      if (k && (k.startsWith('ws_') || k.startsWith('inv_'))) return; // 既存データ保護
    }
    const P = COMPANY_CONFIG.orderPrefix;
    // ……ここで ws_/inv_/projects_saved/client_master_cache を setItem（内容は下表）……
    localStorage.setItem('sample_seeded', '1');
  } catch (e) { console.warn('sample seed skip', e); }
}
```
