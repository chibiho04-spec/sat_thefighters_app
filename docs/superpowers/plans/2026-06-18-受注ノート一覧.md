# 受注ノート一覧ページ 実装計画

> **For agentic workers:** 単一HTMLファイル `案件管理アプリ/index.html` を編集。自動テストは無いため検証は (a)`node`でscript抽出→compileFunction構文チェック、(b)Chrome実機E2E（同期は `window._gasUrlOne=()=>''` でOFF）。読み取り専用ビューなのでデータ書き込みは一切しない。

**Goal:** 会計室に「受注ノート」全画面一覧ページを追加。新規案件作成で登録した案件を、受注番号を先頭に1案件＝1行で表示し、T/E/Pほかで並び替え・検索でき、行クリックでワークシート詳細を開く。

**Architecture:** 既存の `worksheet-list-screen` と同じスライドインオーバーレイ方式。データは `getSavedProjects()`＋`getWsDisplayData(num)`（読み取りのみ）。既存関数は変更せず流用。

**Tech Stack:** 素のHTML/CSS/JS（インライン）、localStorage（読み取りのみ）。

仕様：`docs/superpowers/specs/2026-06-18-受注ノート一覧-design.md`

## 既存コードの要点
- 会計室タブ tab-1 の「書類作成」内に `openWorksheetList()` ボタン（📋ワークシート）。この**上**に新ボタンを置く。
- `getSavedProjects()`：projects_saved配列。各案件 `{num,title,company,date,type,orderType,staff,crew*,memo,receivedAt,...}`。
- `getWsDisplayData(num)`：title/company/date(getWsDateText)/type/orderType を編集値優先で解決（staff/receivedAtは含まない→projectレコードから取る）。
- `getDeletedWS()`：削除済み番号。`parseDateStr(str)`：日付（範囲対応）→[{y,m,d}]。`openWS(num)`：詳細を開く。`escapeHtml`。
- デスクトップ配置CSS：`#xxx-screen,` を3グループ（index.html 約2155 / 約2227 / 約2875、いずれも top:54px・left:200px 系）に追加必須。
- スライドイン開閉：`el.style.transform='translateX(0)'`（開）/`'translateX(100%)'`（閉）。

## Task 2: JS — open/close・収集・並び替え・描画・検索

**Files:** Modify `index.html`

- [ ] **Step 1: 関数を追加**（`closeWorksheetList` の近く等、ワークシート一覧系の隣に）

```js
  function openOrderNoteScreen() {
    renderOrderNoteList();
    document.getElementById('order-note-screen').style.transform = 'translateX(0)';
  }
  function closeOrderNoteScreen() {
    document.getElementById('order-note-screen').style.transform = 'translateX(100%)';
  }

  // 受注ノート1行ぶんの表示データ（読み取りのみ）
  function _orderNoteRow(num) {
    const d = getWsDisplayData(num); // title/company/date/type/orderType
    let proj = {};
    try { proj = (getSavedProjects().find(p => String(p.num) === String(num))) || {}; } catch (e) {}
    return {
      num: String(num),
      title: d.title || '無題',
      company: d.company || '',
      date: d.date || '',
      type: d.type || '',
      orderType: d.orderType || '',
      staff: proj.staff || '',
      receivedAt: proj.receivedAt || ''
    };
  }

  function renderOrderNoteList() {
    const box = document.getElementById('order-note-list');
    if (!box) return;
    const deleted = (typeof getDeletedWS === 'function') ? getDeletedWS() : [];
    // 対象：projects_saved の案件（削除済み除く）
    let nums = [];
    try { nums = getSavedProjects().map(p => String(p.num)).filter(n => n && deleted.indexOf(n) < 0); } catch (e) { nums = []; }
    nums = [...new Set(nums)];
    let rows = nums.map(_orderNoteRow);

    // 検索
    const q = (document.getElementById('order-note-search')?.value || '').trim().toLowerCase();
    if (q) rows = rows.filter(r => [r.num, r.title, r.company, r.staff].join(' ').toLowerCase().indexOf(q) >= 0);

    // 並び替え
    const sort = document.getElementById('order-note-sort')?.value || 'tep';
    const prefRank = n => { const p = String(n).charAt(0).toUpperCase(); return p === 'T' ? 0 : p === 'E' ? 1 : p === 'P' ? 2 : 9; };
    const firstShoot = r => { const a = parseDateStr(r.date); return a.length ? new Date(a[0].y, a[0].m - 1, a[0].d).getTime() : null; };
    rows.sort((a, b) => {
      if (sort === 'tep')          return (prefRank(a.num) - prefRank(b.num)) || a.num.localeCompare(b.num, 'ja');
      if (sort === 'num_desc')     return b.num.localeCompare(a.num, 'ja');
      if (sort === 'num_asc')      return a.num.localeCompare(b.num, 'ja');
      if (sort === 'company')      return String(a.company).localeCompare(String(b.company), 'ja') || a.num.localeCompare(b.num, 'ja');
      if (sort === 'received_desc') { const x = a.receivedAt || '', y = b.receivedAt || ''; if (!x && !y) return 0; if (!x) return 1; if (!y) return -1; return y.localeCompare(x); }
      if (sort === 'shoot_desc')   { const x = firstShoot(a), y = firstShoot(b); if (x == null && y == null) return 0; if (x == null) return 1; if (y == null) return -1; return y - x; }
      return 0;
    });

    document.getElementById('order-note-count').textContent = rows.length + ' 件';
    const esc = (typeof escapeHtml === 'function') ? escapeHtml : (s => String(s));
    const pfxColor = n => { const p = String(n).charAt(0).toUpperCase(); return p === 'T' ? '#2980b9' : p === 'E' ? '#27ae60' : p === 'P' ? '#8e44ad' : '#7f8c8d'; };
    if (!rows.length) {
      box.innerHTML = '<div style="padding:30px 12px;text-align:center;color:#999;font-size:13px;">該当する案件がありません</div>';
      return;
    }
    box.innerHTML = rows.map(r =>
      '<div onclick="openWS(\'' + esc(r.num) + '\')" style="display:flex;align-items:baseline;gap:8px;flex-wrap:wrap;padding:8px 10px;margin-bottom:4px;background:#fff;border:1px solid #e0e4e8;border-left:5px solid ' + pfxColor(r.num) + ';border-radius:4px;cursor:pointer;box-shadow:1px 1px 0 #e8edf1;">' +
        '<span style="font-family:\'DotGothic16\',monospace;font-weight:bold;font-size:14px;color:' + pfxColor(r.num) + ';min-width:92px;flex-shrink:0;">' + esc(r.num) + '</span>' +
        '<span style="font-size:14px;font-weight:bold;color:#333;flex:1;min-width:120px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">' + esc(r.title) + '</span>' +
        '<span style="font-size:11px;color:#666;">🏢' + esc(r.company || '—') + '</span>' +
        (r.date ? '<span style="font-size:11px;color:#888;">📅' + esc(r.date) + '</span>' : '') +
        (r.type ? '<span style="font-size:10px;color:#2980b9;background:#eaf3fb;padding:1px 6px;border-radius:8px;">' + esc(r.type) + '</span>' : '') +
        (r.orderType ? '<span style="font-size:10px;color:#8e44ad;background:#f4ecf7;padding:1px 6px;border-radius:8px;">' + esc(r.orderType) + '</span>' : '') +
        (r.staff ? '<span style="font-size:11px;color:#555;">👤' + esc(r.staff) + '</span>' : '') +
      '</div>'
    ).join('');
  }
```
