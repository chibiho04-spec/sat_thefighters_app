> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Tech Stack:** 単一HTMLファイル（インラインHTML/CSS/JS）、ブラウザのみ。テストは `node --check` 相当（script抽出構文チェック）＋Chrome実機（同期OFF `window._gasUrlOne=()=>''`）。

**設計仕様:** `docs/superpowers/specs/2026-06-19-進行中ダッシュボード-design.md`

```javascript
  // 機材名リスト（ws_ の equip チェック済み × equip_master 名）。ダッシュボード用。
  function getWsEquipNames(num) {
    try {
      const saved  = (typeof loadWS === 'function') ? loadWS(num) : {};
      const equip  = saved.equip || {};
      const master = (typeof loadEquipMaster === 'function') ? loadEquipMaster() : [];
      const names = [];
      master.forEach(item => {
        const sv = equip[item.id];
        if (sv && sv.checked) {
          const q = (sv.qty !== undefined ? sv.qty : item.qty) || 1;
          names.push(item.name + (Number(q) > 1 ? ' ×' + q : ''));
        }
      });
      return names;
    } catch (e) { return []; }
  }

  // ダッシュボード対象＝撮影日が未来・未請求・撮影日程ありの案件（初日昇順）
  function getDashboardCases() {
    let deleted = []; try { deleted = getDeletedWS() || []; } catch (e) {}
    const delSet = new Set(deleted.map(String));
    const now = new Date();
    const todayNum = now.getFullYear() * 10000 + (now.getMonth() + 1) * 100 + now.getDate();
    let projs = []; try { projs = getSavedProjects() || []; } catch (e) {}
    const out = [];
    projs.forEach(p => {
      if (!p || p.num == null || delSet.has(String(p.num))) return;
      const num = String(p.num);
      let d = {}; try { d = getWsDisplayData(num) || {}; } catch (e) { d = {}; }
      const dateStr = (d.date && d.date !== '日程未定') ? d.date : (p.date || '');
      let days = []; try { days = parseDateStr(dateStr) || []; } catch (e) {}
      if (!days.length) return; // 撮影日程なしは除外
      const first = days.reduce((mn, x) => { const v = x.y * 10000 + x.m * 100 + x.d; return v < mn ? v : mn; }, Infinity);
      if (!(first > todayNum)) return; // 撮影日がまだ来ていない（未来）のみ
      let billed = false; try { billed = !!((loadInvStatus(num) || {}).billed); } catch (e) {}
      if (billed) return; // 未請求のみ
      out.push({ num, title: d.title || p.title || '無題の案件', company: d.company || p.company || '',
        dateStr, firstKey: first, type: d.type || '', orderType: d.orderType || '', staff: p.staff || '' });
    });
    out.sort((a, b) => (a.firstKey - b.firstKey) || String(a.num).localeCompare(String(b.num), 'ja', { numeric: true }));
    return out;
  }
```

```javascript
  let _dashTimer = null;
  let _dashReturn = false;

  function updateDashClock() {
    const e = document.getElementById('dash-clock'); if (!e) return;
    const n = new Date();
    e.textContent = [n.getHours(), n.getMinutes(), n.getSeconds()].map(x => String(x).padStart(2, '0')).join(':');
  }

  function renderDashCalendar(cases) {
    const now = new Date();
    const y = now.getFullYear(), m = now.getMonth(), todayD = now.getDate();
    const startDow = new Date(y, m, 1).getDay();
    const daysInMonth = new Date(y, m + 1, 0).getDate();
    const byDay = {};
    cases.forEach(c => {
      let days = []; try { days = parseDateStr(c.dateStr) || []; } catch (e) {}
      days.forEach(dd => { if (dd.y === y && (dd.m - 1) === m) (byDay[dd.d] = byDay[dd.d] || []).push(c); });
    });
    let cells = '';
    for (let i = 0; i < startDow; i++) cells += '<div class="dash-cal-cell dash-cal-empty"></div>';
    for (let d = 1; d <= daysInMonth; d++) {
      const cs = byDay[d] || [];
      const chips = cs.slice(0, 3).map(c =>
        `<div class="dash-cal-chip" onclick="dashOpenWS('${String(c.num).replace(/'/g, "\\'")}')" title="${escapeHtml(c.title)}">${escapeHtml(c.title)}</div>`).join('');
      const more = cs.length > 3 ? `<div class="dash-cal-more">+${cs.length - 3}</div>` : '';
      cells += `<div class="dash-cal-cell${d === todayD ? ' dash-cal-today' : ''}"><div class="dash-cal-num">${d}</div>${chips}${more}</div>`;
    }
    return '<div class="dash-cal-dow"><span class="dow-sun">日</span><span>月</span><span>火</span><span>水</span><span>木</span><span>金</span><span class="dow-sat">土</span></div>'
      + `<div class="dash-cal-grid">${cells}</div>`;
  }

  function renderDashWorksheets(cases) {
    if (!cases.length) return '<div class="dash-empty">これからの現場はありません</div>';
    const PROG = ['受付', '準備中', '撮影中', '請求書作成中'];
    const PROGC = ['#7f8c8d', '#f39c12', '#2980b9', '#16a085'];
    const MAX = 9;
    const tiles = cases.slice(0, MAX).map(c => {
      let prog = 0; try { prog = calcWsProgress(c.num) || 0; } catch (e) {}
      prog = Math.min(Math.max(prog, 0), 3);
      let crew = []; try { crew = getWsCrewList(c.num) || []; } catch (e) {}
      let equip = []; try { equip = getWsEquipNames(c.num) || []; } catch (e) {}
      const crewTxt = crew.length ? escapeHtml(crew.join('　')) : '—';
      const equipTxt = equip.length ? escapeHtml(equip.slice(0, 4).join('／') + (equip.length > 4 ? ` ほか${equip.length - 4}` : '')) : '—';
      const col = (typeof _onPrefixColor === 'function') ? _onPrefixColor(c.num) : '#2980b9';
      const numA = String(c.num).replace(/'/g, "\\'");
      return `<div class="dash-ws-tile" style="border-top-color:${col};" onclick="dashOpenWS('${numA}')">
        <div class="dash-ws-top"><span style="color:${col};font-weight:bold;">${escapeHtml(c.num)}</span><span class="dash-ws-prog" style="background:${PROGC[prog]};">${PROG[prog]}</span></div>
        <div class="dash-ws-date">📅 ${escapeHtml(c.dateStr)}</div>
        <div class="dash-ws-title">${escapeHtml(c.title)}</div>
        <div class="dash-ws-company">${escapeHtml(c.company)}</div>
        <div class="dash-ws-sec">クルー</div><div class="dash-ws-val">${crewTxt}</div>
        <div class="dash-ws-sec">機材</div><div class="dash-ws-val dash-ws-equip">${equipTxt}</div>
      </div>`;
    }).join('');
    const more = cases.length > MAX ? `<div class="dash-ws-more">ほか ${cases.length - MAX} 件</div>` : '';
    return tiles + more;
  }

  function renderDashboard() {
    const el = document.getElementById('dashboard-screen'); if (!el) return;
    const cases = getDashboardCases();
    const now = new Date();
    const we = new Date(); we.setDate(we.getDate() + 7);
    const weekEndNum = we.getFullYear() * 10000 + (we.getMonth() + 1) * 100 + we.getDate();
    const weekCount = cases.filter(c => c.firstKey <= weekEndNum).length;
    const set = (id, v) => { const e = document.getElementById(id); if (e) e.textContent = v; };
    set('dash-count-total', cases.length + '件');
    set('dash-count-week', weekCount + '件');
    set('dash-date', `${now.getMonth() + 1}/${now.getDate()}(${'日月火水木金土'[now.getDay()]})`);
    set('dash-cal-title', `${now.getFullYear()}年 ${now.getMonth() + 1}月`);
    const cal = document.getElementById('dash-cal'); if (cal) cal.innerHTML = renderDashCalendar(cases);
    const ws = document.getElementById('dash-ws'); if (ws) ws.innerHTML = renderDashWorksheets(cases);
  }

  function openDashboard() {
    const el = document.getElementById('dashboard-screen'); if (!el) return;
    el.style.display = 'flex'; el.classList.add('dash-show');
    renderDashboard(); updateDashClock();
    if (_dashTimer) clearInterval(_dashTimer);
    let tick = 0;
    _dashTimer = setInterval(() => { updateDashClock(); if ((++tick % 60) === 0) { try { renderDashboard(); } catch (e) {} } }, 1000);
  }

  function closeDashboard() {
    const el = document.getElementById('dashboard-screen'); if (el) { el.style.display = 'none'; el.classList.remove('dash-show'); }
    if (_dashTimer) { clearInterval(_dashTimer); _dashTimer = null; }
    if (location.hash === '#dashboard') { try { history.replaceState(null, '', location.pathname + location.search); } catch (e) { location.hash = ''; } }
  }

  // タイル/チップから編集へ：ダッシュボードを隠して openWS、closeWS で復帰
  function dashOpenWS(num) {
    const el = document.getElementById('dashboard-screen');
    if (el && el.style.display !== 'none') { el.style.display = 'none'; _dashReturn = true; }
    try { openWS(num); } catch (e) { console.warn('dashOpenWS', e); }
  }
```
